cmake_minimum_required(VERSION 3.23.2)
project(Singularity)
add_executable(Singularity MACOSX_BUNDLE "../../singularity.cpp" "../../singularity.metal")
set_source_files_properties("../../singularity.metal" PROPERTIES LANGUAGE METAL COMPILE_FLAGS "-I \"../..\" -I \"../../gpu\" -std=macos-metal2.4 -Werror -D prayer_major_version=0 -D prayer_middle_version=0 -D prayer_minor_version=1")
set_target_properties(Singularity PROPERTIES BUNDLE True MACOSX_BUNDLE_GUI_IDENTIFIER "com.iristechnologies.Singularity" MACOSX_BUNDLE_BUNDLE_NAME Singularity MACOSX_BUNDLE_BUNDLE_VERSION "0.0.1" MACOSX_BUNDLE_SHORT_VERSION_STRING "0.0.1" MACOSX_BUNDLE_INFO_PLIST "temp.plist")
target_include_directories(Singularity PRIVATE "../..")
set(CMAKE_CONFIGURATION_TYPES Release CACHE STRING "" FORCE)
target_link_libraries(Singularity PRIVATE "-framework AppKit")
target_link_libraries(Singularity PRIVATE "-framework QuartzCore")
target_link_libraries(Singularity PRIVATE "-framework Metal")
set(CMAKE_OSX_ARCHITECTURES "arm64 x86_64")
set(CMAKE_CXX_FLAGS "-I \"../..\" -std=c++20 -fno-rtti -fno-exceptions -O3 -Wall -Wpedantic -Wextra -Werror -pedantic-errors -D egoist_major_version=0 -D egoist_middle_version=0 -D egoist_minor_version=1")
